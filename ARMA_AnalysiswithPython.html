import statsmodels.tsa.stattools as ts
import numpy as np
import pandas as pd

Let's say we come across a time series that we want to model as ARMA. How do we start ?

  1. Get ACF and PACF plots
  2. Try to guess the order of AR and MA through PACF and ACF plots respectively
  3. We may not be complete sure about order, so we try to fit all possibilities and calculate fitness of each model using with respect to chosen metrics
  4. The one with better fit (say based on AIC) is choosen
  5. We can sacrifice a marginally better fit in favour of parsimony

Let's synthesize a ARMA(p,q) process

$X_t = Z_t + 0.7Z_{t-1} + 0.5Z_{t-2}+ 0.6X_{t-1}$
Thus, p=1 and q=2 for this process.

import matplotlib.pyplot as plt
from statsmodels.tsa import arima_process
ma_params = [1,0.7,0.5]
ar_params = [1,-0.6]
x = arima_process.arma_generate_sample(ar=ar_params,ma=ma_params,nsample=100)
plt.figure(figsize=(24,6))
plt.subplot(1,3,1)
plt.title("ARMA Process")
plt.plot(x)

plt.subplot(1,3,2)
acorf_x = ts.acf(x,fft=False,adjusted=False,nlags=20)
plt.bar(x=np.arange(len(acorf_x)),height=acorf_x,width=0.3)
plt.axhline(y=0,color='black')
plt.axhline(y=0.2,color='red',linestyle='--')
plt.axhline(y=-0.2,color='red',linestyle='--')
plt.title("Auto Correlation of ARMA process")

plt.subplot(1,3,3)
apcorf_x = ts.pacf(x,nlags=20)
plt.bar(x=np.arange(len(apcorf_x)),height=apcorf_x,width=0.3)
plt.axhline(y=0,color='black')
plt.axhline(y=0.2,color='red',linestyle='--')
plt.axhline(y=-0.2,color='red',linestyle='--')
plt.title("Partial Auto Correlation of ARMA process")

#plt.plot(x)
plt.show()

ACF indicates that MA process could be of order of 3 (i.e. q=3)
PACF indicates that AR process could be of order of 2 (i.e. p=2)
So, let's try p=(0,1,2) and q=(0,1,2,3)

from statsmodels.tsa.arima.model import ARIMA
models = []
results = []
orders = []
for p in range(3):
    for q in range(4):
        mod = ARIMA(endog=x, order=(p,0,q))
        res = mod.fit()
        models.append(mod)
        results.append(res)
        orders.append((p,q))
        print('Order p=%d,q=%d: AIC : %f , Parameters : p = %s q = %s"'%(p,q,np.round(res.aic,4),str(res.polynomial_ar),str(res.polynomial_ma)))
